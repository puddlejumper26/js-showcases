    {

        for (var i = 0; i < oTab13_7_2.tBodies[0].rows.length; i++) 
        {
            arr[i] = oTab13_7_2.tBodies[0].rows[i];

            return n1 - n2;
        });

        {
            oTab13_7_2.tBodies[0].appendChild(arr[i]);
        }
