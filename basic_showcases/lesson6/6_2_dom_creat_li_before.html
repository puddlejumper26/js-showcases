var oBtn12_2=document.getElementById('btn12_2');
var oUl12_2=document.getElementById('ul12_2');
var oTxt12_2=document.getElementById('text12_2');

oBtn12_2.onclick=function()
{
    var oLi12_2=document. createElement('li') 
    var aLi12_2=oUl12_2.getElementsByTagName('li');
    oLi12_2.innerHTML=oTxt12_2.value;  
            //oUl12_1. appendChild (oLi12_1);	
    oTxt12_2.value='';       
    if (aLi12_2.length > 0)
    {        
        oUl12_2.insertBefore(oLi12_2,aLi12_2[0]); // 直接使用会在IE中报错，所以用一个if
    }
    else
    {
        oUl12_2.appendChild(oLi12_2);
    }
}
