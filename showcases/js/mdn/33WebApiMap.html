Simple Mapquest example

        // 1. The basic part of the example
        var L;
        window.onload = function() {
        
        L.mapquest.key = '0KBzNHAKGqnEAkRTXV4LLy6JMLGRB9ui';
        
        // 'map' refers to a 
 element with the ID map
            var map = L.mapquest.map('map', {
                // center: [53.480759, -2.242631],
                center:[48.137154, 11.576124],
                layers: L.mapquest.tileLayer('hybrid'), //其他选项 hybrid,satellite,dark,light
                zoom: 12
            });
        
            //add control
            map.addControl(L.mapquest.control({position: 'topleft'})); //topright,bottomleft,bottomright
        
            //add customer marker 就是标记点
            L.marker([48.137154, 11.576124], {
                icon: L.mapquest.icons.marker({
                    primaryColor: '#22407F',
                    secondaryColor: '#3B5998',
                    shadow: true,
                    size: 'md',
                    symbol: 'M' //这里的字母可以换，但是主要是一个字母
                })
            }) //显示点击指标的 文本
            .bindPopup('This is Munich!')
            .addTo(map);
            }
